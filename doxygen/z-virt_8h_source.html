19 /*
20  * Memory management routines.
21  *
22  * These routines work best as a *replacement* for malloc/free.
23  *
24  * The string_make() and string_free() routines handle dynamic strings.
25  * A dynamic string is a string allocated at run-time, which should not
26  * be modified once it has been created.
27  *
28  * Note the macros below which simplify the details of allocation,
29  * deallocation, setting, clearing, casting, size extraction, etc.
30  *
31  * The macros MAKE/C_MAKE and KILL/C_KILL have a "procedural" metaphor,
32  * and they actually modify their arguments.
33  *
34  * Note that, for some reason, some allocation macros may disallow
35  * "stars" in type names, but you can use typedefs to circumvent
36  * this. For example, instead of "type **p; MAKE(p,type*);" you
37  * can use "typedef type *type_ptr; type_ptr *p; MAKE(p,type_ptr)".
38  *
39  * Note that it is assumed that "memset()" will function correctly,
40  * in particular, that it returns its first argument.
41  */

78 template <typename T, std::enable_if_t<std::is_trivially_copyable_v<T>, std::nullptr_t> = nullptr>
79 inline T *c_copy_impl(T *p1, T *p2, size_t n)

84 // トリビアルコピーが不可能な型は要素を1つずつコピー代入する
85 template <typename T, std::enable_if_t<!std::is_trivially_copyable_v<T>, std::nullptr_t> = nullptr>
86 inline T *c_copy_impl(T *p1, T *p2, size_t n)
87 {
88  for (size_t i = 0; i < n; i++) {
89  *(p1 + i) = *(p2 + i);
90  }
91  return p1;
92 }
